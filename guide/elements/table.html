    function trAddClass(e) {
        var dataItems = e.sender.dataSource.view();
        for (var j = 0; j < dataItems.length; j++) {
            var distinct = dataItems[j].get("비고");
            var row = e.sender.tbody.find("[data-uid='" + dataItems[j].uid + "']");
            if (distinct == "반려") {
                row.addClass("status-reject");
            } else if (distinct == "대기") {
                row.addClass("status-wait");
            }
        }
    };
