삭제되었습니다.

전환되었습니다.


<button id="viewModel_AlertNohead_btn" data-role="button">Alert (No Heaer)</button>
<div data-role="dialog"
    data-title=""
    data-modal="false"
    data-bind="dialogOpen: isEnabled,
                events: { close: onClose}" 
    id="viewModel_AlertNohead"
    class="alert">
    <div class="content-group">
        <p class="main-text">
            삭제되었습니다.<br/>
            대기건 <span class="point">3</span>건의 가상계좌로<br/>

<script> 
    // Implement custom binder for the 'visible' configuration option
    kendo.data.binders.widget.dialogOpen = kendo.data.Binder.extend({
        init: function (widget, bindings, options) {
            kendo.data.Binder.fn.init.call(this, widget.element[0], bindings, options);
            //헤더없을때
            $('.k-dialog').addClass('no-header'); 
            //높이
            if($('.k-dialog .content-group').prop('scrollHeight') + 60 > $(window).height()) {
                $('.k-dialog').css({"height":"100%"});
            }
        },
        refresh: function () {
            var that = this;
            var value = that.bindings["dialogOpen"].get();
            var dialog = $(that.element).data("kendoDialog");
            if (value) {
                dialog.open();
            } else {
                dialog.close();
            }
            $('#viewModel_AlertNohead_btn').on('click', function() {
                dialog.open();
            })
        }
    });

    var viewModel_AlertNohead = kendo.observable({
        isVisible: true,
        isClosable: true,
        isEnabled: false,
        onClose: function (e) {
            this.set("isVisible", false);
            this.set("isEnabled", true);
        },
        show: function (e) {
            this.set("isVisible", true);
            this.set("isEnabled", false);
        },
    });
    kendo.bind($("#viewModel_AlertNohead"), viewModel_AlertNohead);
</script>
