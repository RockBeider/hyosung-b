        </div>
    </div>
    <div class="right-group">
        <span class="amount-set">
            <span class="amount current">1</span><span class="unit">건</span><span class="dash">/</span>

<div class="optional-box" id="viewModel_listbox">
    <script type="text/x-kendo-template" id="leftboxTemplate">
        <div class="list-box listTransferTo">
            <h3 class="title">#: ProductName #</h3>
        </div>
    </script>
    <script type="text/x-kendo-template" id="rightboxTemplate">
        <div class="list-box listTransferFrom">
            <h3 class="title">#: ProductName #</h3>
            <i class="h-ico close"></i>
        </div>
    </script> 
    <select id="listbox1" 
        data-role="listbox"
        data-text-field="ProductName"
        data-value-field="ProductID"
        data-connect-with="listbox2"
        data-template="leftboxTemplate"
        data-toolbar='{
            tools: ["transferTo", "transferFrom", "transferAllTo", "transferAllFrom", "remove"]
        }'
        data-bind="source: products">
    </select>
    <select id="listbox2" 
        data-role="listbox"
        data-text-field="ProductName"
        data-value-field="ProductID"
        data-connect-with="listbox1"
        data-template="rightboxTemplate"
        >
    </select>
    <script>
        //listbox
        var viewModel_listbox = kendo.observable({
            products: new kendo.data.DataSource({
                schema: {
                    model: {
                        id: "ProductID",
                        fields: {
                            ProductName: { type: "string" },
                            UnitPrice: { type: "number" }
                        }
                    }
                },
                batch: true,
                pageSize: 15,
                transport: {
                    read: {
                        url: "https://demos.telerik.com/kendo-ui/service/products",
                        dataType: "jsonp"
                    }
                }
            })
        });
        kendo.bind($("#viewModel_listbox"), viewModel_listbox);
        $(document).on('change', '#listTransferAllTo input', function(e) {  // 오른쪽으로 전체 옮기기  
            if($('#listTransferAllTo input').is(":checked") == true ) {
                console.log('checked')
                $("#viewModel_listbox .k-listbox-toolbar [data-command='transferAllTo']").trigger('click');
            } else {
                $("#viewModel_listbox .k-listbox-toolbar [data-command='transferAllFrom']").trigger('click');
            }
        });
        $(document).on('click', '#viewModel_listbox .listTransferTo', function(e) {  //오른쪽으로 옮기기
            $("#viewModel_listbox .k-listbox-toolbar [data-command='transferTo']").trigger('click');
        });
        $(document).on('click', '#viewModel_listbox .listTransferFrom', function(e) { //왼쪽으로 옮기기
            $("#viewModel_listbox .k-listbox-toolbar [data-command='transferFrom']").trigger('click');
        });
    </script>
</div>
